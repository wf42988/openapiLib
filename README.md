func strToLong(timeStr string) int64 {
	timeTemplate := "2006-01-02 15:04:05"
	stamp, _ := time.ParseInLocation(timeTemplate, timeStr, time.Local) //使用parseInLocation将字符串格式化返回本地时区时间
	return stamp.Unix() * 1000
}
